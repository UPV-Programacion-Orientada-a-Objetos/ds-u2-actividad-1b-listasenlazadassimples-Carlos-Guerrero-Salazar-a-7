13 * @brief Constructor de la clase ArduinoSerial.
14 * @details Intenta abrir el puerto serial, configura los parámetros de comunicación (9600 baudios, 8N1, no canónico)
15 * y establece la conexión.
16 * @param puerto Puntero a una cadena de caracteres con la ruta del dispositivo serial a utilizar (p. ej., "/dev/ttyUSB0").
17 * @pre Los puertos seriales deben estar configurados correctamente en el sistema operativo.
18 */
19ArduinoSerial::ArduinoSerial(const char* puerto) : conectado(false) {
20 // Intentar abrir el puerto en modo lectura/escritura y sin terminal de control (O_NOCTTY)
21 serial_port = open(puerto, O_RDWR | O_NOCTTY);
22 if (serial_port < 0) {
23 std::cerr << "[ERROR] No se pudo abrir el puerto serial: " << puerto << " (Error: " << errno << ")" << std::endl;
24 return;
25 }
26
27 struct termios tty;
28 std::memset(&tty, 0, sizeof(tty));
29
30 // Obtener atributos actuales
31 if (tcgetattr(serial_port, &tty) != 0) {
32 std::cerr << "[ERROR] Error en tcgetattr" << std::endl;
33 close(serial_port);
34 serial_port = -1;
35 return;
36 }
37
38 // Configurar velocidad (Baud Rate)
39 cfsetispeed(&tty, B9600);
40 cfsetospeed(&tty, B9600);
41
42 // Configuración CFLAGS (Control Mode)
43 tty.c_cflag &= ~PARENB; // Sin paridad
44 tty.c_cflag &= ~CSTOPB; // 1 bit de parada
45 tty.c_cflag &= ~CSIZE; // Limpiar bits de tamaño
46 tty.c_cflag |= CS8; // 8 bits de datos
47 tty.c_cflag &= ~CRTSCTS; // Deshabilitar control de flujo de hardware (RTS/CTS)
48 tty.c_cflag |= CREAD | CLOCAL; // Habilitar recepción y no tomar control del puerto
49
50 // Configuración LFLAGS (Local Mode)
51 tty.c_lflag &= ~ICANON; // Deshabilitar modo canónico (lectura byte a byte)
52 tty.c_lflag &= ~ECHO; // Deshabilitar eco
53 tty.c_lflag &= ~ISIG; // Deshabilitar manejo de señales especiales (INTR, QUIT, etc.)
54
55 // Configuración IFLAGS (Input Mode)
56 tty.c_iflag &= ~(IXON | IXOFF | IXANY); // Deshabilitar control de flujo de software (XON/XOFF)
57 tty.c_iflag &= ~(IGNBRK|BRKINT|PARMRK|ISTRIP|INLCR|IGNCR|ICRNL); // Deshabilitar procesamiento especial de entrada
58
59 // Configuración OFLAGS (Output Mode)
60 tty.c_oflag &= ~OPOST; // Deshabilitar procesamiento de salida
61 tty.c_oflag &= ~ONLCR; // Deshabilitar mapeo NL a CR-NL
62
63 // Configuración CC (Control Characters) - Bloqueo de tiempo
64 tty.c_cc[VTIME] = 10; // Tiempo de espera (en décimas de segundo)
65 tty.c_cc[VMIN] = 0; // Retorno inmediato si no hay datos
66
67 // Aplicar atributos
68 if (tcsetattr(serial_port, TCSANOW, &tty) != 0) {
69 std::cerr << "[ERROR] Error en tcsetattr" << std::endl;
70 close(serial_port);
71 serial_port = -1;
72 return;
73 }
74
75 conectado = true;
76 std::cout << "[OK] Puerto serial " << puerto << " abierto correctamente" << std::endl;
77}

80/**
81 * @brief Lee una línea de datos terminada en salto de línea ('\n') desde el puerto serial.
82 * @details Esta función utiliza una lectura bloqueante (con timeout por `VTIME`) y almacena
83 * los bytes hasta encontrar un '\n' o alcanzar el límite de 255 caracteres.
84 * @return Un puntero estático a la cadena de caracteres con la línea leída, o `nullptr` si la conexión no está activa o si ocurre un error de lectura.
85 * @pre Debe haber una conexión activa (`conectado == true`) con el puerto serial.
86 */
87char* ArduinoSerial::leerLinea() {
88 if (!conectado) {
89 return nullptr;
90 }
91
92 static char linea[256];
93 int idx = 0;
94 char buf;
95
96 while (idx < 255) {
97 int n = read(serial_port, &buf, 1);
98
99 if (n > 0) {
100 if (buf == '\n') {
101 linea[idx] = '\0';
102 return linea;
103 } else if (buf != '\r') {
104 linea[idx++] = buf;
105 }
106 } else if (n < 0 && errno != EAGAIN && errno != EWOULDBLOCK) {
107 // Error real de lectura
108 std::cerr << "[ERROR] Error leyendo del puerto serial" << std::endl;
109 return nullptr;
110 }
111 }
112 linea[255] = '\0';
113 return linea;
114}
